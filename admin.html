<script>
let channels = JSON.parse(localStorage.getItem('app_channels')) || [
{ name: "قناة 1", url: "https://www.youtube.com/embed/live_stream?channel=YOUR_CHANNEL_ID" },
{ name: "قناة 2", url: "https://www.youtube.com/embed/live_stream?channel=YOUR_CHANNEL_ID" }
];

const container = document.getElementById('channels-editor');

function renderEditor() {
container.innerHTML = '';
channels.forEach((ch, index) => {
container.innerHTML += `
<div class="channel-item">
<button class="delete-btn" onclick="removeChannel(${index})">حذف</button>
<label>اسم القناة ${index + 1}:</label>
<input type="text" id="name_${index}" value="${ch.name}">

<label>رابط البث (Embed URL):</label>
<input type="text" id="url_${index}" value="${ch.url}">
</div>
`;
});
}

function addChannel() {
channels.push({ name: `قناة ${channels.length + 1}`, url: "" });
renderEditor();
}

function removeChannel(index) {
channels.splice(index, 1);
renderEditor();
}

function saveAll() {
channels.forEach((ch, index) => {
ch.name = document.getElementById(`name_${index}`).value;
ch.url = document.getElementById(`url_${index}`).value;
});

localStorage.setItem('app_channels', JSON.stringify(channels));
alert('تم حفظ القنوات بنجاح في MCR7!');
}

renderEditor();
</script>
