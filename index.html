<script>
let channels = JSON.parse(localStorage.getItem('app_channels')) || [
{ name: "قناة 1", url: "https://www.youtube.com/embed/live_stream?channel=YOUR_CHANNEL_ID" },
{ name: "قناة 2", url: "https://www.youtube.com/embed/live_stream?channel=YOUR_CHANNEL_ID" },
{ name: "قناة 3", url: "https://www.youtube.com/embed/live_stream?channel=YOUR_CHANNEL_ID" }
];

const channelsList = document.getElementById('channels-list');
const player = document.getElementById('main-player');
const channelTitle = document.getElementById('current-channel-title');

function loadChannels() {
channelsList.innerHTML = '';
channels.forEach((ch, index) => {
const btn = document.createElement('div');
btn.className = 'channel-btn' + (index === 0 ? ' active' : '');
btn.innerText = ch.name;
btn.onclick = () => {
document.querySelectorAll('.channel-btn').forEach(b => b.classList.remove('active'));
btn.classList.add('active');
player.src = ch.url;
channelTitle.innerText = ch.name;
};
channelsList.appendChild(btn);
});

if (channels.length > 0) {
player.src = channels[0].url;
channelTitle.innerText = channels[0].name;
}
}

loadChannels();
</script>
